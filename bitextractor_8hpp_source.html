175 indices.cend(),
176 [ &n_items ]( uint32_t index ) -> bool {

180 throw BitException( "Cannot extract item at the index " + std::to_string( *find_res ),

182 }

253 void extractMatchingFilter( Input in_archive,
254 const tstring& out_dir,
255 FilterPolicy policy,
256 const function< bool( const tstring& ) >& filter ) const {
257 BitInputArchive input_archive( *this, in_archive );
258
259 vector< uint32_t > matched_indices;
260 const bool should_extract_matched_items = policy == FilterPolicy::Include;
261 // Searching for files inside the archive that match the given filter
262 for ( const auto& item : input_archive ) {
263 const bool item_matches = filter( item.path() );
264 if ( item_matches == should_extract_matched_items ) {
265 /* The if-condition is equivalent to an exclusive NOR (negated XOR) between
266 * item_matches and should_extract_matched_items.
267 * In other words, it is true only if the current item either:
268 * - matches the filter, and we must include any matching item; or
269 * - doesn't match the filter, and we must exclude those that match. */
270 matched_indices.push_back( item.index() );
271 }
272 }
273
274 if ( matched_indices.empty() ) {
275 throw BitException( "Cannot extract items", make_error_code( BitError::NoMatchingItems ) );
276 }
277
278 input_archive.extract( out_dir, matched_indices );
279 }
280
281 void extractMatchingFilter( Input in_archive,
282 vector< byte_t >& out_buffer,
283 FilterPolicy policy,
284 const function< bool( const tstring& ) >& filter ) const {
285 BitInputArchive input_archive( *this, in_archive );
286
287 const bool should_extract_matched_item = policy == FilterPolicy::Include;
288 // Searching for files inside the archive that match the given filter
289 for ( const auto& item : input_archive ) {
290 const bool item_matches = filter( item.path() );
291 if ( item_matches == should_extract_matched_item ) {
292 /* The if-condition is equivalent to an exclusive NOR (negated XOR) between
293 * item_matches and should_extract_matched_item. */
294 input_archive.extract( out_buffer, item.index() );
295 return;
296 }
297 }
298
299 throw BitException( "Failed to extract items", make_error_code( BitError::NoMatchingItems ) );
300 }
