36using underlying_type_t = typename std::underlying_type< E >::type;
37
38template< typename E >
39inline constexpr auto to_underlying( E e ) noexcept -> underlying_type_t< E > {
40 return static_cast< underlying_type_t< E > >( e );
41}

43inline constexpr FormatFeatures operator|( FormatFeatures lhs, FormatFeatures rhs ) noexcept {
44 return static_cast< FormatFeatures >( to_underlying( lhs ) | to_underlying( rhs ) );
